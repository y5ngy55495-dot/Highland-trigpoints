                <div class="popup-cell"><div class="popup-cell-label">Order</div><div class="popup-cell-value">${p.o}</div></div>
                <div class="popup-cell"><div class="popup-cell-label">Grid</div><div class="popup-cell-value">${p.nn}</div></div>
                <div class="popup-cell"><div class="popup-cell-label">Coords</div><div class="popup-cell-value">${p.la.toFixed(4)}, ${p.lo.toFixed(4)}</div></div>
            </div>
            <span class="popup-status ${statusClass}">${statusText}</span>
            <button class="bag-action-btn ${isBagged ? 'bagged' : ''}" onclick="toggleBag('${p.n.replace(/'/g, "\'")}')">${bagText}</button>
        </div>
    `);

    marker.pillarData = p;
    allMarkers.push(marker);
    markers.addLayer(marker);
});

map.addLayer(markers);

function updateView() {
    markers.clearLayers();
    let visible = 0;

    allMarkers.forEach(m => {
        const p = m.pillarData;
        const isDestroyed = p.d === '1';
        const isBagged = baggedTrigs.has(p.n);

        // Apply filter
        let show = true;
        if (currentFilter === 'standing' && isDestroyed) show = false;
        if (currentFilter === 'destroyed' && !isDestroyed) show = false;
        if (currentFilter === 'unbagged' && isBagged) show = false;
        if (currentFilter === 'bagged' && !isBagged) show = false;

        if (show) {
            visible++;
            // Update style
            const color = isBagged ? '#34d399' : (isDestroyed ? '#f87171' : getHeightColor(p.h));
            m.setStyle({
                radius: isBagged ? 8 : (isDestroyed ? 5 : 7),
                fillColor: color, color: isBagged ? '#34d399' : (isDestroyed ? '#f87171' : '#fff'),
                weight: isBagged ? 3 : (isDestroyed ? 1 : 2.5),
                opacity: isDestroyed && !isBagged ? 0.4 : 0.9,
                fillOpacity: isDestroyed && !isBagged ? 0.25 : 0.85
            });
            markers.addLayer(m);
        }
    });

    updateCount();
}

function toggleBag(name) {
    if (baggedTrigs.has(name)) {
        baggedTrigs.delete(name);
    } else {
        baggedTrigs.add(name);
    }
    try { localStorage.setItem('highlandsTrigBagged', JSON.stringify([...baggedTrigs])); } catch(e) {}

    // Update popup button
    const popup = document.querySelector('.bag-action-btn');
    if (popup) {
        const isBagged = baggedTrigs.has(name);
        popup.classList.toggle('bagged', isBagged);
        popup.textContent = isBagged ? '✓ Bagged' : 'Mark as Bagged';
    }

    updateView();
}

function setFilter(filter) {
    currentFilter = filter;
    document.querySelectorAll('.bottom-btn').forEach(b => b.classList.remove('active'));
    document.getElementById('btn-' + filter).classList.add('active');
    updateView();
}

function updateCount() {
    const count = baggedTrigs.size;
    document.getElementById('bag-count').textContent = count + ' / 1,605';
}

// Search
const searchInput = document.getElementById('search-input');
const searchResults = document.getElementById('search-results');

searchInput.addEventListener('input', function() {
    const q = this.value.toLowerCase();
    if (q.length < 2) { searchResults.style.display = 'none'; return; }

    const matches = pillarData.filter(p => p.n.toLowerCase().includes(q) || p.nn.toLowerCase().includes(q)).slice(0, 10);
    searchResults.innerHTML = '';

    if (matches.length) {
        matches.forEach(m => {
            const h = parseFloat(m.h) || 0;
            const div = document.createElement('div');
            div.className = 'result';
            div.innerHTML = `<span class="result-name">${m.n}</span><span>${h.toFixed(0)}m</span>`;
            div.addEventListener('click', () => {
                map.setView([m.la, m.lo], 15);
                searchResults.style.display = 'none';
                searchInput.value = '';
                allMarkers.forEach(marker => {
                    if (marker.pillarData.n === m.n) {
                        markers.zoomToShowLayer(marker, () => marker.openPopup());
                    }
                });
            });
            searchResults.appendChild(div);
        });
        searchResults.style.display = 'block';
    } else {
        searchResults.style.display = 'none';
    }
});

document.addEventListener('click', e => {
    if (!e.target.closest('.search-wrap')) searchResults.style.display = 'none';
});

// Fit bounds
const bounds = L.latLngBounds(pillarData.map(p => [p.la, p.lo]));
map.fitBounds(bounds, { padding: [60, 60] });

updateCount();
